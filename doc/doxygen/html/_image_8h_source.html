218 template <class PixelType>
219 bool Image<PixelType>::allocate_mem()
220 {
221  int h, pos;
222  // sizeByte elem
223  PixelType *buf1D;
224 
225  // (height + heightUV * 2) rows, Y (0) - U (height) - V (height + heightUV)
226  PixelType **buf2D;
227 
228  release_mem();
229 
230  if ((buf1D = (PixelType *)malloc(m_sizeByte)) == NULL) return false;
231 
232  if ((buf2D = (PixelType **)malloc((m_height + m_heightUV * 2) * sizeof(PixelType *))) == NULL)
233  {
234  free(buf1D);
235  return false;
236  }
237  // fill (picsizeY) elem = 0 for Y
238  memset(&buf1D[0], 0, m_picsizeY * sizeof(PixelType));
239  // fill (picsizeUV * 2) elem = 128 for U, V
240  memset(&buf1D[m_picsizeY], 128, m_picsizeUV * 2 * sizeof(PixelType));//Owieczka set it
241 
242  Y = buf2D;
243  U = &(buf2D[m_height]);
244  V = &(buf2D[m_height + m_heightUV]);
245 
246  // fill Y, U, V from buf1D
247  for (h = pos = 0; h < m_height; h++, pos += m_width)
248  Y[h] = &(buf1D[pos]);
249  for (h = 0; h < m_heightUV; h++, pos += m_widthUV)
250  U[h] = &(buf1D[pos]);
251  for (h = 0; h < m_heightUV; h++, pos += m_widthUV)
252  V[h] = &(buf1D[pos]);
253 
254  m_comp[0] = Y;
255  m_comp[1] = U;
256  m_comp[2] = V;
257 
258  m_buffer1D = Y[0]; // buf1D;
259 
260  return true;

373 {
374  int h, w, cH, cW;
375  int ir, ig, ib;
376  for (h = cH = 0; h < m_height; h++)
377  {
378  cH = h >> 1;
379  if ((cH << 1) >= h) cH--; //owieczka
380  for (w = cW = 0; w < m_width; w++, cW++)
381  {
382  for (size_t i = 0; i < 2; i++)
383  {
384  ib = (int)(src->Y[h][w] + 1.772*(src->U[cH][cW] - 127) + 0.5);
385  ig = (int)(src->Y[h][w] - 0.344*(src->U[cH][cW] - 127) - 0.714*(src->V[cH][cW] - 127) + 0.5);
386  ir = (int)(src->Y[h][w] + 1.402*(src->V[cH][cW] - 127) + 0.5);
387 
388  m_comp[0][h][w] = ImageTools::CLIP3(ib, 0, MaxTypeValue<PixelType>() - 1);
389  m_comp[1][h][w] = ImageTools::CLIP3(ig, 0, MaxTypeValue<PixelType>() - 1);
390  m_comp[2][h][w] = ImageTools::CLIP3(ir, 0, MaxTypeValue<PixelType>() - 1);
391 
392  if (i == 0) w++;
393  }
